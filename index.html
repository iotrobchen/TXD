async function getCCTV() { const client_id = '你的ID'; const client_secret = '你的Secret'; // 使用 corsproxy.io 避開前端阻擋問題 const proxy = "https://corsproxy.io/?"; const auth_url = `${proxy}https://tdx.transportdata.tw/auth/realms/TDXConnect/protocol/openid-connect/token`; const api_url = `${proxy}https://tdx.transportdata.tw/api/basic/v2/Road/Traffic/CCTV/City/Taipei?$top=10&$format=JSON`; try { console.log("正在取得 Token..."); const authRes = await fetch(auth_url, { method: 'POST', headers: { 'content-type': 'application/x-www-form-urlencoded' }, body: new URLSearchParams({ grant_type: 'client_credentials', client_id: client_id, client_secret: client_secret }) }); if (!authRes.ok) throw new Error(`Token 取得失敗: ${authRes.status}`); const authData = await authRes.json(); const token = authData.access_token; console.log("Token 取得成功，正在抓取 CCTV 資料..."); const apiRes = await fetch(api_url, { headers: { 'Authorization': `Bearer ${token}` } }); const data = await apiRes.json(); renderCCTV(data); } catch (err) { console.error('詳細錯誤訊息:', err); document.getElementById('display').innerText = '失敗原因: ' + err.message; } } function renderCCTV(data) { const container = document.getElementById('display'); container.innerHTML = ''; data.forEach(item => { const div = document.createElement('div'); div.className = 'cctv-item'; // 台北市的影像網址有些需要補後綴或處理，這裡直接嘗試顯示 div.innerHTML = `

${item.CCTVName}

`; container.appendChild(div); }); }
